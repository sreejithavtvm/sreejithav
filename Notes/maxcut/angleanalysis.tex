\begin{enumerate}

\item Region B: $\trans{\vec r} \vec u_i \geq 0$ and $\trans{\vec r} \vec u_j \geq 0$
\item Region C: $\trans{\vec r} \vec u_i < 0$ and $\trans{\vec r} \vec u_j \geq 0$
\item Region D: $\trans{\vec r} \vec u_i < 0$ and $\trans{\vec r} \vec u_j < 0$
\end{enumerate}
